Cash Register high

Design a cash register drawer function checkCashRegister() that accepts purchase price as the first argument (price), payment as the second argument (cash), and cash-in-drawer (cid) as the third argument.
cid is a 2D array listing available currency.
The checkCashRegister() function should always return an object with a status key and a change key.
Return {status: "INSUFFICIENT_FUNDS", change: []} if cash-in-drawer is less than the change due, or if you cannot return the exact change.
Return {status: "CLOSED", change: [...]} with cash-in-drawer as the value for the key change if it is equal to the change due.
Otherwise, return {status: "OPEN", change: [...]}, with the change due in coins and bills, sorted in highest to lowest order, as the value of the change key.

# Current Unit Amount
1 Penny $0.01 (PENNY)
2 Nickel $0.05 (NICKEL)

4 Quarter $0.25 (QUARTER)
5 Dollar $1 (ONE)
6 Five Dollars $5 (FIVE)
7 Ten Dollars $10 (TEN)
8 Twenty Dollars $20 (TWENTY)
9 One-hundred Dollars $100 (ONE HUNDRED)

function checkCashRegister(price, cash, cid) {
    let dev = cash - price;
    let obj = {status:"", change:[]}

        case "NICKEL": bill = 0.05; break;
        case "DIME": bill = 0.1; break;
        case "QUARTER": bill = 0.25; break;
        case "ONE": bill = 1; break;

        case "TEN": bill = 10; break;
        case "TWENTY": bill = 20; break;
        case "ONE HUNDRED": bill = 100; break;

        if(dev >= cid[i][1] && cid[i][1]!==0){
        dev = dev.toFixed(2)-cid[i][1];
        value = cid[i][1];
        newArr= [cid[i][0],value];
        obj.change.push(newArr)
        cid[i][1] = 0;
        }else if(dev < cid[i][1]){
        let condition = dev/bill;
        let need = Math.floor(condition);
        if(condition >= 1 && cid[i][1] >= need*bill){
        
            dev = dev.toFixed(2) - (need*bill).toFixed(2);
            value = Number((need*bill).toFixed(2));
            newArr= [cid[i][0],value];
            obj.change.push(newArr)
            cid[i][1] = cid[i][1] - (need*bill).toFixed(2);
        }else if(cid[i][1] < need*bill){

            value = cid[i][1];
